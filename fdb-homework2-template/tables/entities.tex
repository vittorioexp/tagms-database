    Employee & Represents data of an employee who works in the company and needs access to the system &
    \begin{itemize}
        \vspace{-1em}
        \item TAX\_number:   TAX code of the employee, text
        \item First\_name:   name of the employee, text
        \item Last\_name:   surname of the employee, text
        \item Phone\_number:   phone number (prefix included) of the employee, text
        \item Email\_address:   email address of the employee, text
        \item Birth\_date:   birthdate of the employee, datetime
        \item Hiring\_date:   hiring date of the employee, datetime
        \item Still\_working: flag used to know if employee is still working for the company, boolean
        \item Role\_ID:   role identifier, int
        \item Department\_ID:   department identifier, int
    \end{itemize}
    &  TAX\_number \\\hline

    Role & Represents data on the role of employees who work in the company &
    \begin{itemize}
        \vspace{-1em}
        \item ID:   role identifier, serial
        \item Name:   name of the role, text
        \item Description:   technical description of the role, text
    \end{itemize}
    &  ID \\\hline

    Department & Represents data on the departments in which employees work &
    \begin{itemize}
        \vspace{-1em}
        \item ID:   department identifier of the company, serial
        \item Name:   name of the department, text
        \item Description:   description of the department's function, text
    \end{itemize}
    &  ID \\\hline

    Customer & Represents data about a customer of the company &
    \begin{itemize}
        \vspace{-1em}
        \item Email\_address:   email address of the customer, text
        \item Customer\_name:   name of the customer, text
        \item Phone\_number:   phone number (prefix included) of the customer, text
        \item Address:   billing address of the customer, text
        \item Registration\_date:   customer registration date in the database, datetime
    \end{itemize}
    &  Email\_address \\\hline

    \begin{itemize}

        \item ID:   contract identifier, serial
        \item Description:   description of the contract, text
        \item Contract\_date:    date of signature of the contract with the supplier, datetime
        \item Expiration\_date:    expiration date of the contract with the supplier, datetime
        \item Supplier\_ID:   supplier identifier, int        % length=11
        \item Employee\_ID:   employee identifier, text
    \end{itemize}
    &  ID \\\hline

    Supplier & Represents data about a supplier of the company &
    \begin{itemize}
        \vspace{-1em}
        \item VAT\_number:   VAT number of the supplier company, int
        \item Supplier\_name:   name of the supplier company, text
        \item Phone\_number:   phone number (prefix included) of the supplier company, text
        \item Email\_address:   email address of the supplier company, text
        \item Address:   address of the supplier company, text
        \item Registration\_date:   recording date of the supplier company, datetime
    \end{itemize}
    &  VAT\_number \\\hline

    %External constraint: if we want we can do a load analysis
    Order & Represents the order placed by a salesman for a customer &
    \begin{itemize}
        \vspace{-1em}
        \item ID:   order identifier, serial
        \item Order\_date:   date in which the order has been processed, datetime
        \item Order\_paid:   status of the payment, boolean
        \item Net\_price:   total net amount including discount (VAT excluded), float    %net_price = for each lot { lot_price * (1 - discount / 100) }
        \item Taxes:   total amount of taxes to be payed, float                     %taxes = for each lot { lot_price * (vat / 100) }
                                                                                    %Total_amount = net_price + taxes
    \end{itemize}
    &  ID \\\hline

    %External constraints: discount[0-100] set by the company. VAT may vary over time. vat [0-100]
    Lot & Represents the inventory of the company, containing products and items &
    \begin{itemize}
        \vspace{-1em}
        \item ID:   lot identifier, serial
        \item Order\_ID:   identifier of the associated order, serial     %TODO: is it correct?
        \item Product\_ID:   identifier of the included product, serial
        \item Product\_quantity:   quantity of the included product, int
        \item Package\_ID:   package identifier related to the specific lot, serial
        \item Package\_quantity:   lot packaging quantity, int
        \item Expiration\_date:   expiration date of the included products, datetime
        \item Lot\_price:   lot price (without the discount) at the time of sale, float
        \item Lot\_discount:   percentage of discount of a lot, int
        \item VAT:   value added tax (percentage) at the time of sale, int
    \end{itemize}
    &  ID \\\hline

    Product & Represents the final product that is marketed &
    \begin{itemize}
        \vspace{-1em}
        \item ID:   product identifier, serial
        \item Name:   name of the product, text
        \item Description:   description of the product, text
        \item Energy:   energy nutritional value (kJ per 100ml), float
        \item Fat:   fat nutritional value (grams per 100ml), float
        \item Saturated fatty acids:   saturated fatty acids nutritional value (grams per 100ml), float
        \item Carbohydrates:   carbohydrates nutritional value (grams per 100ml), float
        \item Sugar:   sugar nutritional value (grams per 100ml), float
        \item Protein:   protein nutritional value (grams per 100ml), float
        \item Salt:   salt nutritional value (grams per 100ml), float
        \item Volume:   volume of the product in milliliters, int
        \item Net\_weight:   net weight of the product in grams, int
        \item Package\_weight:   package weight of the product in grams, int
        \item Production\_cost:   cost of the production for a product, float
        \item Inflation:   price increase factor, float                         %product_price = inflation x production_cost
        \item Product\_Category\_ID:   identifier of the product category, int
    \end{itemize}
    &  ID \\\hline

    % External constraint: There is no expiration date because we assume that item are used very fast. We take this into account in the expiration date of the product

        \item Name:   name of the item, text
        \item Description:   description of the item, text

    \end{itemize}
    &  ID \\\hline
